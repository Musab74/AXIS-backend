[코드 핵심]
import pandas as pd
import logging

logging.basicConfig(filename='error.log', level=logging.ERROR)

try:
    df = pd.read_csv('sales_weekly.csv')
    prev = pd.read_csv('prev_sales_weekly.csv')
    
    current = df.groupby('product')['amount'].sum().reset_index()
    current.columns = ['product', 'total_amount']
    current = current.sort_values('total_amount', ascending=False)
    current['rank'] = range(1, len(current)+1)
    
    prev_sum = prev.groupby('product')['amount'].sum().reset_index()
    prev_sum.columns = ['product', 'prev_amount']
    
    result = current.merge(prev_sum, on='product', how='left')
    result['change_rate'] = ((result['total_amount'] - result['prev_amount']) / result['prev_amount'] * 100).round(1)
    
    top3 = result.head(3)
    top3.to_csv('report.csv', index=False, encoding='utf-8-sig')
except Exception as e:
    logging.error(f"처리 오류: {e}")

import pandas as pd

def calculate_monthly_sales(filepath):
    df = pd.read_csv(filepath)
    df['date'] = pd.to_datetime(df['date'])
    df['month'] = df['date'].dt.month
    
    monthly = df.groupby('month')['amount'].mean()  # 오류1: mean→sum
    monthly = monthly.reset_index()
    monthly.columns = ['month', 'total']
    
    monthly['formatted'] = monthly['total'].apply(lambda x: f"{x:,.0f}원")
    
    monthly.to_csv('monthly_report.csv', index=False)
    print(f"처리 완료: {len(monthly)}개월")
    return monthly

result = calculate_monthly_sales('sales.csv')  # 오류2: 파일 미존재 시 에러 처리 없음
print(result.sort_values('total'))  # 오류3: ascending=False 누락 (내림차순이어야 함)

[코드 핵심]
import os, requests, logging
from datetime import date

API_KEY = os.environ.get('WEATHER_API_KEY')
if not API_KEY:
    raise ValueError("환경변수 WEATHER_API_KEY가 설정되지 않았습니다.")

url = f"https://api.openweathermap.org/data/2.5/weather?q=Seoul&appid={API_KEY}&units=metric&lang=kr"

try:
    resp = requests.get(url, timeout=10)
    resp.raise_for_status()
    data = resp.json()
    weather = data['weather'][0]['description']
    temp = round(data['main']['temp'])
    humidity = data['main']['humidity']
    msg = f"{date.today()} 오늘의 날씨: {weather}, {temp}°C, 습도 {humidity}%"
    with open('weather_notice.txt','w',encoding='utf-8') as f:
        f.write(msg)
except requests.RequestException as e:
    logging.error(f"API 오류: {e}")

import json

    response = requests.get(url)  # 오류2: 인증 헤더 누락, timeout 미설정

def save_data(data, filename):

import pandas as pd
import os, logging, glob

logging.basicConfig(filename='batch.log', level=logging.INFO)
files = glob.glob('reports/*.xlsx')
all_data = []

for f in files:
    try:
        df = pd.read_excel(f, sheet_name='요약')
        all_data.append(df[['부서명','매출','비용','이익']])
        logging.info(f"처리 완료: {f}")
    except Exception as e:
        logging.error(f"처리 실패: {f} - {e}")

if all_data:
    result = pd.concat(all_data, ignore_index=True)
    total = pd.DataFrame([{'부서명':'합계','매출':result['매출'].sum(),'비용':result['비용'].sum(),'이익':result['이익'].sum()}])
    result = pd.concat([result, total], ignore_index=True)
    result.to_excel('consolidated_report.xlsx', index=False)

import pandas as pd

def filter_data(filepath, min_amount):
    df = pd.read_excel(filepath)
    
    filtered = df[df['amount'] > min_amount]  # 오류1: >= 이어야 함 (이상, 초과 아님)
    
    filtered['ratio'] = filtered['amount'] / df['total']  # 오류2: df 대신 filtered 사용해야 함 (인덱스 불일치)
    
    filtered.to_excel('filtered_output.xlsx')
    
    summary = filtered.groupby('dept')['amount'].sum()
    print(f"부서 수: {len(summary)}")
    print(f"평균 금액: {summary.mean():,.0f}원")
    
    return filtered

result = filter_data('data.xlsx', 1000000)

import re, json
from collections import Counter

pattern = r'(\S+) .+ \[(\d{2}/\w+/\d{4}):(\d{2}):\d{2}:\d{2}\] "\w+ (\S+) .+" (\d{3})'

with open('access_log.txt','r') as f:
    lines = f.readlines()

status_counts = Counter()
path_404 = Counter()
hourly = Counter()

for line in lines:
    m = re.match(pattern, line)
    if m:
        ip, date, hour, path, status = m.groups()
        status_counts[status] += 1
        hourly[hour] += 1
        if status == '404':
            path_404[path] += 1

report = {
    'status_counts': dict(status_counts),
    'top_404_paths': [p for p,_ in path_404.most_common(5)],
    'hourly_counts': dict(sorted(hourly.items()))
}

with open('log_report.json','w',encoding='utf-8') as f:
    json.dump(report, f, ensure_ascii=False, indent=2)

[제공 코드]
import os

def search_keyword(folder, keyword):
    results = []
    
    for filename in os.listdir(folder):
        filepath = folder + filename  # 오류1: os.path.join 미사용 (OS별 경로 구분자 문제)
        
        with open(filepath, 'r') as f:  # 오류2: encoding 미지정 (한글 파일 UnicodeDecodeError)
            for i, line in enumerate(f):
                if keyword in line:
                    results.append({
                        'file': filename,
                        'line': i,  # 오류3: i는 0부터 시작, 사용자에게는 1부터 표시해야 함
                        'content': line.strip()
                    })
    
    print(f"'{keyword}' 검색 결과: {len(results)}건")
    return results

matches = search_keyword('./logs/', '에러')
for m in matches:
    print(f"  {m['file']} 라인{m['line']}: {m['content']}")
